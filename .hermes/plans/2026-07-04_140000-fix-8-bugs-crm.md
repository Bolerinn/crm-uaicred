### Task 1: Fix Lixeira "Excluir Todos" — deleção silenciosa

**Problema:** `excluirTodosLixeira()` (linha ~3327) chama `sb.from('clientes').delete().eq('id', id)` mas falha silenciosamente (RLS: só o criador pode hard-delete). O toast mostra "0 processo(s) excluído(s)" ou o count errado. Os itens continuam na lixeira.

**Files:**
- Modify: `C:\Users\Prime04D\crm-uaicred\index.html` (função `excluirTodosLixeira` ~3327-3339)

**Step 1: Corrigir `excluirTodosLixeira`**

Substituir a função atual por versão com feedback de erro e re-render após cada deleção bem-sucedida:

```javascript
async function excluirTodosLixeira() {
  const checks = document.querySelectorAll('.lixeira-check:checked');
  if (checks.length === 0) { mostrarToast('Nenhum item selecionado'); return; }
  if (!confirm('ATENÇÃO: ' + checks.length + ' processo(s) serão EXCLUÍDOS PERMANENTEMENTE. Continuar?')) return;
  let count = 0;
  let erros = 0;
  for (const cb of checks) {
    const id = parseInt(cb.value);
    const { error } = await sb.from('clientes').delete().eq('id', id);
    if (!error) {
      count++;
      clientesLixeira = clientesLixeira.filter(c => c.id !== id);
    } else {
      erros++;
      console.error('Falha ao excluir permanentemente id=' + id + ':', error.message);
    }
  }
  if (count > 0) mostrarToast(count + ' processo(s) excluído(s) ✕');
  if (erros > 0) mostrarToast('⚠️ ' + erros + ' processo(s) NÃO puderam ser excluídos (apenas o criador pode excluir permanentemente)');
  renderizarTabelaLixeira();
  // Recarregar do banco pra garantir sincronia
  await carregarLixeira();
}
```

### Task 2: Fix Editar parceiro duplica na Aba 1

**Problema:** `salvarEdicaoParceiro()` (linha ~2450) quando o nome muda, empurra um registro legacy com o nome antigo. Isso cria entrada duplicada na lista de parceiros.

**Files:**
- Modify: `C:\Users\Prime04D\crm-uaicred\index.html` (função `salvarEdicaoParceiro` ~2450-2491)

**Step 1: Corrigir `salvarEdicaoParceiro`**

Remover o push de legacy entry quando o nome muda. Em vez disso, se houver processos usando o nome antigo, atualizar o campo `indicacao` nesses processos para o novo nome:

```javascript
function salvarEdicaoParceiro() {
  const nomeOriginal = document.getElementById('editParceiroNomeOriginal').value;
  const novoNome = document.getElementById('editParceiroNome').value.trim();
  const novoTipo = document.getElementById('editParceiroTipo').value;
  if (!novoNome) { mostrarToast('Preencha o nome'); return; }
  
  const lista = carregarIndicacoes();
  const idx = lista.findIndex(i => i.nome === nomeOriginal);
  if (idx < 0) { mostrarToast('Parceiro não encontrado'); return; }
  
  // Check duplicate only if name actually changed
  if (novoNome.toLowerCase() !== nomeOriginal.toLowerCase() &&
      lista.some(i => i.nome.toLowerCase() === novoNome.toLowerCase())) {
    mostrarToast('⚠️ Já existe um parceiro com esse nome');
    return;
  }
  
  // Count how many processes use this partner
  const emUso = clientes.filter(c => (c.indicacao || '').trim() === nomeOriginal).length;
  
  const novaImobiliaria = document.getElementById('editImobiliaria').value.trim();

  lista[idx].nome = novoNome;
  lista[idx].tipo = novoTipo;
  lista[idx].imobiliaria = novaImobiliaria;
  
  // Se nome mudou e há processos usando o nome antigo, atualiza o indicacao neles
  if (novoNome !== nomeOriginal && emUso > 0) {
    // Atualizar clientes no array local (será persistido no blur)
    clientes.forEach(c => {
      if ((c.indicacao || '').trim() === nomeOriginal) {
        c.indicacao = novoNome;
      }
    });
    // Também atualizar no banco
    (async () => {
      const { error } = await sb.from('clientes').update({ indicacao: novoNome }).eq('indicacao', nomeOriginal);
      if (!error) await carregarDados();
    })();
  }
  
  salvarIndicacoes(lista);
  popularDatalistIndicacoes();
  fecharModalEditarParceiro();
  abrirModalParceiros(); // refresh
  if (novoNome !== nomeOriginal && emUso > 0) {
    mostrarToast('✅ Parceiro atualizado (' + emUso + ' processos atualizados)');
  } else {
    mostrarToast('✅ Parceiro atualizado');
  }
}
```

function removerParceiro(nome) {
  const emUso = clientes.filter(c => (c.indicacao || '').trim() === nome).length;
  
  if (emUso > 0) {
    mostrarToast('⚠️ "' + nome + '" está vinculado a ' + emUso + ' processo(s). Remova ou altere a indicação nos processos primeiro.');
    return;
  }
  
  if (!confirm('Remover "' + nome + '" da lista de parceiros?')) return;
  const lista = carregarIndicacoes();
  const novaLista = lista.filter(i => i.nome !== nome);
  salvarIndicacoes(novaLista);
  popularDatalistIndicacoes();
  abrirModalParceiros();
  mostrarToast('🗑️ Parceiro removido da lista');
}
